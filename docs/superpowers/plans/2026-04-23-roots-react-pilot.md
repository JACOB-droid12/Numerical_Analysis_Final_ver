# Roots React Pilot Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build an isolated `roots-react/` Vite + React + TypeScript + Tailwind pilot for the Roots Workbench while leaving the current static calculator and `roots/` app intact.

**Architecture:** Create a standalone React app under `roots-react/`. Load copied legacy engine files from `roots-react/public/legacy/`, call `window.RootEngine` through a typed adapter, and render the answer-first Roots workflow with focused React components and TypeScript contracts.

**Tech Stack:** Vite, React, TypeScript, Tailwind CSS v4 via `@tailwindcss/vite`, npm scripts, existing legacy numerical engines.

```ts
const LEGACY_SCRIPTS = [
  '/legacy/math-engine.js',
  '/legacy/calc-engine.js',
  '/legacy/expression-engine.js',
  '/legacy/root-engine.js',
] as const;

let loadPromise: Promise<void> | null = null;

function loadScript(src: string): Promise<void> {
  return new Promise((resolve, reject) => {
    const existing = document.querySelector<HTMLScriptElement>(
      `script[data-legacy-engine="${src}"]`,
    );

    if (existing?.dataset.loaded === 'true') {
      resolve();
      return;
    }

    const script = existing ?? document.createElement('script');
    script.src = src;
    script.defer = false;
    script.dataset.legacyEngine = src;

    script.addEventListener('load', () => {
      script.dataset.loaded = 'true';
      resolve();
    });
    script.addEventListener('error', () => {
      reject(new Error(`Failed to load legacy engine script: ${src}`));
    });

    if (!existing) {
      document.head.appendChild(script);
    }
  });
}

export function loadLegacyEngines(): Promise<void> {
  if (!loadPromise) {
    loadPromise = LEGACY_SCRIPTS.reduce(
      (chain, src) => chain.then(() => loadScript(src)),
      Promise.resolve(),
    ).then(() => {
      if (!window.RootEngine) {
        throw new Error('RootEngine did not initialize after legacy scripts loaded.');
      }
    });
  }

  return loadPromise;
}
```

```ts
import { METHOD_BY_NAME } from '../config/methods';
import type { IterationRow, RootMethod, RootRunResult } from '../types/roots';

const EMPTY = 'Not calculated yet';

export function methodLabel(method: RootMethod | undefined): string {
  if (!method) return EMPTY;
  return METHOD_BY_NAME.get(method)?.label ?? method;
}

export function formatValue(value: unknown, digits = 12): string {
  if (value == null) return 'N/A';
  if (typeof value === 'number') {
    if (!Number.isFinite(value)) return String(value);
    return Number(value.toPrecision(digits)).toString();
  }
  if (typeof value === 'string') return value;
  if (typeof value === 'object' && 'approx' in value && typeof value.approx === 'number') {
    return formatValue(value.approx, digits);
  }
  if (typeof value === 'object' && 'machine' in value && typeof value.machine === 'number') {
    return formatValue(value.machine, digits);
  }
  return String(value);
}

export function stopReasonLabel(reason: string | null | undefined, method?: RootMethod): string {
  const map: Record<string, string> = {
    'iteration-limit': 'Completed the requested iterations',
    'iteration-cap': 'Stopped at the safety cap',
    'tolerance-reached': 'Reached the requested tolerance',
    'tolerance-already-met': 'Starting interval already meets the tolerance',
    'endpoint-root': 'An endpoint is already the root',
    'exact-zero': method === 'fixedPoint' ? 'The iteration reached an exact fixed point' : 'Reference value is exactly zero',
    'machine-zero': 'Machine value is zero or near zero',
    'invalid-starting-interval': 'Not a valid starting bracket',
    'invalid-bracket': 'The interval does not bracket a sign change',
    'discontinuity-detected': 'Stopped at a discontinuity or singularity',
    'singularity-encountered': 'Function evaluation failed during the iteration',
    'non-finite-evaluation': 'Function evaluation returned a non-finite value',
    'derivative-zero': 'Derivative is zero, so the method cannot continue',
    stagnation: 'The method stalled because the denominator is near zero',
    diverged: 'Iteration diverged',
    'diverged-step': 'Step grew too quickly',
    'step-small-residual-large': 'Step is small but residual remains large',
    'retained-endpoint-stagnation': 'Same endpoint retained too long',
    'cycle-detected': 'Iteration entered a cycle',
    'invalid-input': 'Invalid input',
  };

  return reason ? map[reason] ?? reason : EMPTY;
}

export function stoppingText(run: RootRunResult | null): string {
  if (!run?.stopping) return EMPTY;
  const stopping = run.stopping;
  if (stopping.kind === 'epsilon') {
    const toleranceLabel = stopping.toleranceType ? `${stopping.toleranceType} ` : '';
    const tried = stopping.iterationsRequired ?? stopping.actualIterations ?? run.rows?.length ?? 0;
    return `${toleranceLabel}epsilon = ${stopping.input}, iterations tried = ${tried}`;
  }
  if (run.method === 'bisection') {
    return `n = ${stopping.input}, epsilon <= ${formatValue(stopping.epsilonBound)}`;
  }
  return `n = ${stopping.input}, final |error| = ${formatValue(run.summary?.error)}`;
}

export function interpretationText(run: RootRunResult | null): string {
  if (!run?.summary) return 'Run the method to see a short interpretation.';
  const reason = run.summary.stopReason;
  if (reason === 'iteration-limit') {
    return 'You stopped after the requested iterations, so this is the answer for the fixed-n version of the problem.';
  }
  if (reason === 'tolerance-reached') {
    return 'The method met the requested tolerance under the current precision settings.';
  }
  if (reason === 'endpoint-root') {
    return 'One endpoint is already the root, so no iteration is needed.';
  }
  if (reason === 'exact-zero' || reason === 'machine-zero') {
    return run.method === 'fixedPoint'
      ? 'The iteration landed on a fixed point under the current precision rule.'
      : 'The function value is zero or machine-zero at the reported approximation.';
  }
  if (reason === 'cycle-detected') {
    return 'The fixed-point iteration repeated a cycle instead of settling to one value.';
  }
  if (reason === 'iteration-cap') {
    return 'The safety cap was reached before the requested tolerance was met.';
  }
  return stopReasonLabel(reason, run.method);
}

export function nextActionText(run: RootRunResult | null): string {
  if (!run?.summary) return 'Run the method to see the next recommended action.';
  const reason = run.summary.stopReason;
  if (reason === 'iteration-limit') return 'Need a smaller error? Increase n or switch to tolerance mode.';
  if (reason === 'tolerance-reached') return 'Copy the answer or open the evidence if you need to show work.';
  if (reason === 'invalid-bracket') return 'Choose an interval where the endpoint signs differ.';

  if (reason === 'iteration-cap') return 'Change the starting value or method if the table is not improving.';
  return 'Review the diagnostics and iteration table before changing inputs.';
}

  const approx = formatValue(run.summary?.approximation, 18);
  const count = run.rows?.length ?? 0;

  if (run.method === 'newton') {
    return [

      run.stopping?.kind === 'epsilon'
```

```ts
import type {
  AngleMode,
  DecisionBasis,
  MachineMode,
  MethodFormState,
  RootMethod,
  RootRunResult,
  SignDisplay,
  StoppingKind,
  ToleranceType,
} from '../types/roots';

function requireRootEngine() {
  if (!window.RootEngine) {
    throw new Error('RootEngine is not loaded yet.');
  }
  return window.RootEngine;
}

function machine(fields: MethodFormState, prefix: string) {
  return {
    k: Number(fields[`${prefix}-k`]),
    mode: (fields[`${prefix}-mode`] || 'round') as MachineMode,
  };
}

function stopping(fields: MethodFormState, prefix: string) {
  return {
    kind: (fields[`${prefix}-stop-kind`] || 'iterations') as StoppingKind,
    value: fields[`${prefix}-stop-value`] || '4',
    toleranceType: fields[`${prefix}-tolerance-type`] as ToleranceType | undefined,
  };
}

export function runRootMethod(
  method: RootMethod,
  fields: MethodFormState,
  angleMode: AngleMode,
): RootRunResult {
  const rootEngine = requireRootEngine();

  if (method === 'bisection') {
    return rootEngine.runBisection({
      expression: fields['root-bis-expression'],
      interval: { a: fields['root-bis-a'], b: fields['root-bis-b'] },
      machine: machine(fields, 'root-bis'),
      stopping: stopping(fields, 'root-bis'),
      decisionBasis: (fields['root-bis-decision-basis'] || 'machine') as DecisionBasis,
      signDisplay: (fields['root-bis-sign-display'] || 'both') as SignDisplay,
      angleMode,
    });
  }

  if (method === 'newton') {
    return rootEngine.runNewtonRaphson({
      expression: fields['root-newton-expression'],
      dfExpression: fields['root-newton-df'],
      x0: fields['root-newton-x0'],
      machine: machine(fields, 'root-newton'),
      stopping: stopping(fields, 'root-newton'),
      angleMode,
    });
  }

  if (method === 'secant') {
    return rootEngine.runSecant({
      expression: fields['root-secant-expression'],
      x0: fields['root-secant-x0'],
      x1: fields['root-secant-x1'],
      machine: machine(fields, 'root-secant'),
      stopping: stopping(fields, 'root-secant'),
      angleMode,
    });
  }

  if (method === 'falsePosition') {
    return rootEngine.runFalsePosition({
      expression: fields['root-fp-expression'],
      interval: { a: fields['root-fp-a'], b: fields['root-fp-b'] },
      machine: machine(fields, 'root-fp'),
      stopping: stopping(fields, 'root-fp'),
      decisionBasis: (fields['root-fp-decision-basis'] || 'machine') as DecisionBasis,
      signDisplay: (fields['root-fp-sign-display'] || 'both') as SignDisplay,
      angleMode,
    });
  }

  if (method === 'fixedPoint') {
    return rootEngine.runFixedPoint({
      gExpression: fields['root-fpi-expression'],
      x0: fields['root-fpi-x0'],
      machine: machine(fields, 'root-fpi'),
      stopping: stopping(fields, 'root-fpi'),
      angleMode,
    });
  }

  throw new Error(`Unsupported root method: ${method}`);
}

export function errorMessage(error: unknown): string {
  return error instanceof Error
    ? error.message
    : 'The root calculation could not finish.';
}
```

```ts
import { useCallback, useEffect, useMemo, useState } from 'react';
import { createDefaultFormState, METHOD_CONFIGS } from '../config/methods';

import { errorMessage, runRootMethod } from '../lib/rootEngineAdapter';
import type {
  AngleMode,
  RootMethod,
  RootRunResult,
  WorkbenchStatus,

export function useRootsWorkbench() {
  const [activeMethod, setActiveMethod] = useState<RootMethod>('bisection');
  const [angleMode, setAngleMode] = useState<AngleMode>('deg');
  const [forms, setForms] = useState(() => createDefaultFormState());
  const [runs, setRuns] = useState<Partial<Record<RootMethod, RootRunResult>>>({});
  const [status, setStatus] = useState<WorkbenchStatus>({
    kind: 'loading',
    message: 'Loading root engine...',
  });
  const [evidenceExpanded, setEvidenceExpanded] = useState(false);

  useEffect(() => {
    let active = true;

    loadLegacyEngines()
      .then(() => {
        if (active) {
          setStatus({ kind: 'idle', message: 'Ready.' });
        }
      })
      .catch((error) => {
        if (active) {
          setStatus({ kind: 'error', message: errorMessage(error) });
        }
      });

    return () => {
      active = false;
    };
  }, []);

  const activeConfig = useMemo(
    () => METHOD_CONFIGS.find((config) => config.method === activeMethod) ?? METHOD_CONFIGS[0],
    [activeMethod],
  );

  const activeForm = forms[activeMethod];
  const activeRun = runs[activeMethod] ?? null;

  const updateField = useCallback(
    (method: RootMethod, fieldId: string, value: string) => {
      setForms((current) => ({
        ...current,
        [method]: {
          ...current[method],
          [fieldId]: value,
        },
      }));
      setRuns((current) => ({ ...current, [method]: undefined }));
      setStatus({ kind: 'idle', message: 'Inputs changed. Run again to update the answer.' });
      setEvidenceExpanded(false);
    },
    [],
  );

  const runActiveMethod = useCallback(() => {
    try {
      const run = runRootMethod(activeMethod, forms[activeMethod], angleMode);
      setRuns((current) => ({ ...current, [activeMethod]: run }));
      setStatus({ kind: 'ready', message: 'Answer ready.' });
      setEvidenceExpanded(false);
    } catch (error) {
      setRuns((current) => ({ ...current, [activeMethod]: undefined }));
      setStatus({ kind: 'error', message: errorMessage(error) });
      setEvidenceExpanded(false);
    }
  }, [activeMethod, angleMode, forms]);

  const toggleAngleMode = useCallback(() => {
    setAngleMode((current) => (current === 'deg' ? 'rad' : 'deg'));
    setRuns({});
    setEvidenceExpanded(false);
    setStatus({ kind: 'idle', message: 'Angle mode changed. Re-run to update trig values.' });
  }, []);

  const setMethod = useCallback((method: RootMethod) => {
    setActiveMethod(method);
    setStatus({ kind: 'idle', message: 'Ready.' });
    setEvidenceExpanded(false);
  }, []);

  return {
    activeConfig,
    activeForm,
    activeMethod,
    activeRun,
    angleMode,
    evidenceExpanded,
    forms,
    methodConfigs: METHOD_CONFIGS,
    runs,
    setEvidenceExpanded,
    setMethod,
    status,
    toggleAngleMode,
    updateField,
    runActiveMethod,
  };
}
```

```tsx
  function insertSymbol(text: string) {
    const fieldId = config.expressionFieldId;
    const input = expressionRef.current;
    const current = formState[fieldId] ?? '';
    const start = input?.selectionStart ?? current.length;
    const end = input?.selectionEnd ?? start;
    const next = `${current.slice(0, start)}${text}${current.slice(end)}`;
    onChange(config.method, fieldId, next);
    window.requestAnimationFrame(() => {
      input?.focus();
      input?.setSelectionRange(start + text.length, start + text.length);
    });
  }

  const visibleFields = config.fields.filter((field) => !field.when || field.when(formState));
  const primary = visibleFields.find((field) => field.id === config.expressionFieldId);
  const details = visibleFields.filter((field) => field.id !== config.expressionFieldId && !field.advanced);
  const advanced = visibleFields.filter((field) => field.advanced);

  return (
    <section aria-label="Problem setup" className="space-y-5">
      <div>
        <p className="text-xs font-semibold uppercase tracking-wide text-emerald-800">
          Problem setup
        </p>
        <h2 className="text-2xl font-semibold text-stone-950">{config.label}</h2>
        <p className="mt-1 text-sm text-stone-600">{config.details}</p>
      </div>

      {primary ? (
        <label className="block">
          <span className="mb-2 block text-sm font-semibold text-stone-700">
            {primary.label}
          </span>
          <input
            ref={expressionRef}
            value={formState[primary.id] ?? ''}
            onChange={(event) => onChange(config.method, primary.id, event.target.value)}
            placeholder={primary.placeholder}
            className="w-full rounded-lg border border-stone-300 bg-white px-4 py-3 text-lg text-stone-950 outline-none focus:border-emerald-700 focus:ring-2 focus:ring-emerald-100"
          />
        </label>
      ) : null}

      <SymbolInsertBar onInsert={insertSymbol} />

      <div className="grid gap-3 sm:grid-cols-2 xl:grid-cols-3">
        {details.map((field) => (
          <label key={field.id} className="block">
            <span className="mb-1 block text-sm font-semibold text-stone-700">
              {field.label}
            </span>
            {field.kind === 'select' ? (
              <select
                value={formState[field.id] ?? field.defaultValue}
                onChange={(event) => onChange(config.method, field.id, event.target.value)}
                className="w-full rounded-lg border border-stone-300 bg-white px-3 py-2 text-stone-950"
              >
                {field.options?.map((option) => (
                  <option key={option.value} value={option.value}>
                    {option.label}
                  </option>
                ))}
              </select>
            ) : (
              <input
                type={field.kind === 'number' ? 'number' : 'text'}
                value={formState[field.id] ?? ''}
                onChange={(event) => onChange(config.method, field.id, event.target.value)}
                placeholder={field.placeholder}
                className="w-full rounded-lg border border-stone-300 bg-white px-3 py-2 text-stone-950"
              />
            )}
          </label>
        ))}
      </div>

      {advanced.length ? (
        <details className="rounded-lg border border-stone-300 bg-stone-50 p-4">
          <summary className="cursor-pointer text-sm font-semibold text-stone-800">
            Bracket sign options
```

```tsx
import type { IterationRow, RootRunResult } from '../types/roots';

function xValue(row: IterationRow, method: RootRunResult['method']) {
  if (method === 'bisection' || method === 'falsePosition') return Number(row.c);
  if (method === 'fixedPoint') return Number(row.gxn);
  return Number(row.xNext);
}

interface ConvergenceGraphProps {
  run: RootRunResult;
}

export function ConvergenceGraph({ run }: ConvergenceGraphProps) {
  const points = (run.rows ?? [])
    .map((row) => ({ iteration: row.iteration, x: xValue(row, run.method) }))
    .filter((point) => Number.isFinite(point.x));

  if (!points.length) {
    return (
      <section className="rounded-lg border border-stone-300 bg-white p-4">
        <h3 className="text-base font-semibold text-stone-950">Iterate path</h3>
        <p className="mt-2 text-sm text-stone-600">No iteration data for graph.</p>
      </section>
    );
  }

  const width = 420;
  const height = 190;
  const left = 36;
  const top = 16;
  const plotW = 360;
  const plotH = 130;
  const min = Math.min(...points.map((point) => point.x));
  const max = Math.max(...points.map((point) => point.x));
  const span = max - min || 1;
  const xCoord = (index: number) => left + (index / Math.max(points.length - 1, 1)) * plotW;
  const yCoord = (value: number) => top + plotH - ((value - min) / span) * plotH;
  const path = points
    .map((point, index) => `${index === 0 ? 'M' : 'L'} ${xCoord(index).toFixed(1)} ${yCoord(point.x).toFixed(1)}`)
    .join(' ');

  return (
    <section className="rounded-lg border border-stone-300 bg-white p-4">
      <h3 className="text-base font-semibold text-stone-950">Iterate path</h3>
      <svg viewBox={`0 0 ${width} ${height}`} role="img" aria-label="Iterate values by iteration" className="mt-3 w-full">
        <rect x={left} y={top} width={plotW} height={plotH} rx="6" fill="#f5f5f4" />
        <path d={path} fill="none" stroke="#047857" strokeWidth="3" />
        {points.map((point, index) => (
          <circle key={`${point.iteration}-${index}`} cx={xCoord(index)} cy={yCoord(point.x)} r="4" fill="#064e3b">
            <title>{`i=${point.iteration}, x=${point.x}`}</title>
          </circle>
        ))}
        <text x={left + plotW / 2} y="176" textAnchor="middle" fontSize="12" fill="#57534e">
          Iteration
        </text>
      </svg>
    </section>
  );
}
```
